182@dataclass
183class FurnaceRow:
184    note: Note
185    octave: int
186    instrument: int
187    volume: int
188    effects: List[Tuple[int, int]] = field(default_factory=list)
189
190    def as_clipboard(self) -> str:
191        """
192        Renders the selected row in Furnace clipboard format (without header!)
193
194        :return: Furnace clipboard data
195        """
196        note_maps = {
197            Note.Cs: "C#",
198            Note.D_: "D-",
199            Note.Ds: "D#",
200            Note.E_: "E-",
201            Note.F_: "F-",
202            Note.Fs: "F#",
203            Note.G_: "G-",
204            Note.Gs: "G#",
205            Note.A_: "A-",
206            Note.As: "A#",
207            Note.B_: "B-",
208            Note.C_: "C-",
209        }
210        if self.note == Note.OFF:
211            note_str = "OFF"
212        elif self.note == Note.OFF_REL:
213            note_str = "==="
214        elif self.note == Note.REL:
215            note_str = "REL"
216        elif self.note == Note.__:
217            note_str = "..."
218        else:
219            note_str = "%s%d" % (note_maps[self.note], self.octave)
220
221        vol = ".." if self.volume==0xffff else "%02X" % self.volume
222        ins = ".." if self.instrument==0xffff else "%02X" % self.instrument
223
224        rep_str = "%s%s%s"
225
226        for fx in self.effects:
227            cmd, val = fx
228            cmd_str = ".." if cmd == 0xffff else "%02X" % cmd
229            val_str = ".." if val == 0xffff else "%02X" % val
230            rep_str += "%s%s" % (cmd_str, val_str)
231
232        return rep_str % (
233            note_str,
234            ins, vol
235        ) + "|"
236
237    def __str__(self) -> str:
238        if self.note == Note.OFF:
239            note_str = "OFF"
240        elif self.note == Note.OFF_REL:
241            note_str = "==="
242        elif self.note == Note.REL:
243            note_str = "///"
244        elif self.note == Note.__:
245            note_str = "---"
246        else:
247            note_str = "%s%d" % (self.note, self.octave)
248
249        vol = "--" if self.volume==0xffff else "%02x" % self.volume
250        ins = "--" if self.instrument==0xffff else "%02x" % self.instrument
251
252        rep_str = "row data: %s %s %s"
253
254        for fx in self.effects:
255            cmd, val = fx
256            cmd_str = "--" if cmd == 0xffff else "%02x" % cmd
257            val_str = "--" if val == 0xffff else "%02x" % val
258            rep_str += " %s%s" % (cmd_str, val_str)
259
260        return "<" + rep_str % (
261            note_str,
262            ins, vol
263        ) + ">"

267class FurnacePattern:

269    index: int = 0
270    subsong: int = 0

275        """

277

280        return "org.tildearrow.furnace - Pattern Data\n0\n" + "\n".join([x.as_clipboard() for x in self.data])
